3-3. Collections :

Python code: 
"""
There are four collection data types in Python :
List       : ordered and changeable.            Allows duplicate members.
Tuple      : ordered and unchangeable.          Allows duplicate members.
Set        : unordered and unindexed.           No duplicate members.
Dictionary : unordered, changeable and indexed. No duplicate members.
the index of values start from [0]
There are some difference between each other,
we will mention it later in Module 3. Object
"""
# List
py_list_1 = ['a', 'b', 'c']
py_list_2 = ['e', 'f', 'g']
print(py_list_1)
print(len(py_list_1)) # .length
print(','.join(py_list_1)) # toString()

py_list_3 = py_list_1 + py_list_2 # concat()
print(py_list_3)
print(py_list_3.pop())
print(py_list_3)
py_list_3.append('h') # push()
print(py_list_3)
py_list_3.remove('c') # remove a item
print(py_list_3.index('h')) # indexOf()

print(py_list_3[2])
py_list_3[2]='d'
print(py_list_3[2])
for i in py_list_3:
    print(i)

# copy
py_list_4 = py_list_3 # reference to py_list_3
py_list_3[0] = 'i' # change py_list_3
print(py_list_3)
print(py_list_4) # changes automatically made in py_list_4

py_list_5 = py_list_3.copy() # copy a py_list_3
py_list_3[0] = 'j' # change py_list_3
print(py_list_3)
print(py_list_5) # changes doesn't apply to py_list_5

# Tuple
py_tuple = ('a', 'b', 'c')
print(py_tuple)
print(len(py_tuple))
print(py_tuple[0])
for i in py_tuple:
    print(i)
del py_tuple # delete the tuple

# tuple is unchangeable, so you cannot remove items from it
py_tuple[3] = 'd' # error example, this line will cause error
print(py_tuple)


# Set
py_set = {'a', 'b', 'c'}
print(py_set)
print(len(py_set))
for i in py_set:
    print(i)
py_set.add('d') # like append but unordered
print(py_set)
py_set.update(["e", "f", "g"]) # add multiple items
print(py_set)
py_set.remove('a')
# py_set.remove('a') # this line will cause error, because the 'a' does not exist
print(py_set)
py_set.discard('b') # if the item to remove does not exist
py_set.discard('b') # discard() will NOT raise an error
print(py_set) 
py_set.clear() # empty the set, but not delete
print(py_set) 
del py_set # delete the set

# set is unindexed
print(py_set[0]) # error example, this line will cause error

# Dictionary
person = {
    'name' : 'Julia Chen',
    'birth' : '1995/05/20',
    'mobile' : '0975976521'
    }
print(person)
print(len(person))
for i in person:
    print(i) # get keys
print(person['name']) # get a value
person["birth"] = '1985/10/05' # change value
print(person.get('birth')) # get a value
for i in person.values():
    print(i) # get values
for i in person.items():
    print(i) # get keys and values
person['age'] = 20 # add a key and value
del person['mobile'] # remove a key and value
print(person)
person.clear() # empty the dictionay
print(person)
del person # delete the dictionay
